        {\curso} - {\faculdade}

\newcommand{\folhaderosto}{

		\autor

		\MakeUppercase{\textbf{\titulo}\if\subtitulo\empty\else:\fi}

		\textbf{\subtitulo}

		\vspace{1em}

		\numerovolume

		\vspace{1em}

		\hfill
		\begin{minipage}[t]{.6\textwidth}
			\singlespacing
			\descricao

			{\bigskip}Professora: {\professor}
		\end{minipage}

		\vfill

		\cidade

		\ano

		\skippage
	\end{center}
}
